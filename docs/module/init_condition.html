public subroutine init_cond(zx, zy, zz, tt, uu, vv, ww, ge, g1, g2, zxr, zyr, zzr, ttr, ur, vr, wr, ts)

$ ! superpostion of input NetCDF file and user defined fields $ ! for user defined fields you can use the functions in this module like single_planewave ... $ allocate(zx0(iktx,ikty,iktzp)) $ allocate(zy0(iktx,ikty,iktzp)) $ allocate(zz0(iktx,ikty,iktzp)) $ allocate(tt0(iktx,ikty,iktzp)) $ call ncreadin(zx,zy,zz,tt,ts) $ $ ! superposing the wave field and background QG $ zx = zx + zx0 $ zy = zy + zy0 $ zz = zz + zz0 $ tt = tt + tt0 $ $ ! free the memory for auxilliary variables $ deallocate(zx0,zy0,zz0,tt0)
